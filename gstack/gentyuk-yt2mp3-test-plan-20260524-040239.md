def test_sanitize_empty_title_falls_back_to_video_id():
    assert sanitize("", "dQw4w9WgXcQ") == "dQw4w9WgXcQ"
    assert sanitize("   ", "dQw4w9WgXcQ") == "dQw4w9WgXcQ"

### U2: sanitize Cyrillic + byte truncation
```python
def test_sanitize_preserves_cyrillic():
    s = sanitize("Пятница - Я слышу тебя", "abc")
    assert s == "Пятница - Я слышу тебя"  # exactly preserved

def test_sanitize_truncates_to_200_bytes():
    long_cyr = "Я" * 200  # 400 bytes
    s = sanitize(long_cyr, "abc")
    assert len(s.encode("utf-8")) <= 200
    # decode should still be valid UTF-8 (not split mid-char)
    s.encode("utf-8").decode("utf-8")  # raises if split
```

### U3: sanitize path traversal
```python
def test_sanitize_strips_path_separators():
    assert "/" not in sanitize("a/b/c", "x")
    assert "\\" not in sanitize("a\\b\\c", "x")
    assert ".." not in sanitize("../../etc/passwd", "x")

def test_sanitize_strips_control_chars():
    assert "\x00" not in sanitize("hello\x00world", "x")
    for c in range(0x00, 0x20):
        assert chr(c) not in sanitize(f"a{chr(c)}b", "x")
```

### I1: concurrent SQLite writes
```python
def test_concurrent_inserts_no_lock_errors(tmp_db_path):
    """Five threads each insert 20 rows. Must succeed without OperationalError."""
    def writer(thread_id):
        for i in range(20):
            with db.get_conn(tmp_db_path) as conn:
                db.insert_download(conn, video_id=f"t{thread_id}-{i}", ...)

    threads = [Thread(target=writer, args=(t,)) for t in range(5)]
    [t.start() for t in threads]
    [t.join() for t in threads]

    with db.get_conn(tmp_db_path) as conn:
        count = conn.execute("SELECT COUNT(*) FROM downloads").fetchone()[0]
    assert count == 100  # 5 threads × 20 rows, none lost
```

### I2: queue submits + completes
```python
def test_queue_processes_all_jobs_with_max_workers_3(tmp_db_path, monkeypatch):
    # Mock downloader to return immediately
    monkeypatch.setattr("yt2mp3.downloader.download", lambda url, **kw: Result(status="success", ...))
    q = JobQueue(max_workers=3, db_path=tmp_db_path)
    job_ids = [q.submit(f"https://youtu.be/{i}") for i in range(10)]
    q.wait_all(timeout=5)
    with db.get_conn(tmp_db_path) as conn:
        rows = conn.execute("SELECT COUNT(*) FROM downloads").fetchone()[0]
    assert rows == 10
    assert all(q.get_job(jid) is None for jid in job_ids)  # all removed from in-memory
```

### I3: silent worker crash
```python
def test_worker_exception_logged_and_marked_failed(tmp_db_path, monkeypatch, caplog):
    def crashing_download(*args, **kw):
        raise RuntimeError("boom")
    monkeypatch.setattr("yt2mp3.downloader.download", crashing_download)
    q = JobQueue(max_workers=1, db_path=tmp_db_path)
    job_id = q.submit("https://youtu.be/x")
    q.wait_all(timeout=2)

    # Job removed from in-memory
    assert q.get_job(job_id) is None
    # Logged
    assert any("boom" in rec.message for rec in caplog.records)
    # Marked failed in DB
    with db.get_conn(tmp_db_path) as conn:
        row = conn.execute("SELECT status, error FROM downloads WHERE job_id=?", (job_id,)).fetchone()
    assert row["status"] == "failed"
    assert "boom" in row["error"]
```

### I5: /file/{id} endpoint

def test_file_endpoint_nonexistent_id(client):
    resp = client.get("/file/99999")
    assert resp.status_code == 404

def test_file_endpoint_serves_valid_file(client, tmp_path, tmp_db_path):
    file_path = tmp_path / "test.mp3"
    file_path.write_bytes(b"ID3\x03" + b"\x00" * 100)
    with db.get_conn(tmp_db_path) as conn:
        conn.execute("INSERT INTO downloads (file_path, status) VALUES (?, 'success')",
                     (str(file_path),))
    row_id = conn.lastrowid
    resp = client.get(f"/file/{row_id}")
    assert resp.status_code == 200
    assert resp.content.startswith(b"ID3")

def test_file_endpoint_missing_file_returns_410(client, tmp_db_path):
    with db.get_conn(tmp_db_path) as conn:
        conn.execute("INSERT INTO downloads (file_path, status) VALUES ('/nonexistent.mp3', 'success')")
    row_id = conn.lastrowid
    resp = client.get(f"/file/{row_id}")
    assert resp.status_code == 410  # Gone, not 500

def test_file_endpoint_rejects_path_traversal(client, tmp_db_path):
    # Path traversal via DB-stored path
    with db.get_conn(tmp_db_path) as conn:
        conn.execute("INSERT INTO downloads (file_path, status) VALUES ('../../etc/passwd', 'success')")
    row_id = conn.lastrowid
    resp = client.get(f"/file/{row_id}")
    # is_relative_to check should refuse
    assert resp.status_code in (403, 410)

## Notes

- Smoke test marker: `@pytest.mark.online` registered in `conftest.py`. Run normally with `uv run pytest`, exclude with `uv run pytest -m 'not online'`.
- `tmp_db` fixture: per-test SQLite file with all PRAGMAs applied.
- Mock seam: `yt_dlp.YoutubeDL.extract_info` and `yt_dlp.YoutubeDL.download` patched in I2/I3/I4 to avoid real network.
- All thread-related tests use generous `timeout=N` since slow CI machines can flake at low timeouts.
